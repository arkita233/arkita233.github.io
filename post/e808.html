#反转链表 - 遍历


/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        
        ListNode *res = nullptr;
        ListNode *tmp = head;
        while(tmp)
        {
            ListNode *t = tmp->next;
            
            tmp->next=res;
            res = tmp;
            
            tmp = t;
            
        }
        
        return res;
        
    }
};

#反转链表 - 递归

ListNode reverse(ListNode head) {
    if (head.next == null) return head;

    ListNode last = reverse(head.next);
    head.next.next = head;
    head.next = null;
    return last;
}

根据数据删除链表 - 遍历 1.检查 2.删除除头节点以外符合的 3.检查头结点是否要删除

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) {
        
        ListNode *cur = head;
        
        
        if(!head)
            return NULL;
        if(!head->next && head->val == val)
            return NULL;
        if(!head->next && head->val != val)
            return head;
        
        while(cur !=NULL)
        {
            
            if(cur->next && cur->next->val == val && cur->next->next)
                cur->next = cur->next->next;
            else if(cur->next && cur->next->val == val && !cur->next->next)
                cur->next = NULL;
            else
                cur=cur->next;    
            
            
        }
        
        if(head->val==val)
            head = head->next;
        
        return head;

        
    }
};

根据数据删除链表 - 递归

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
public ListNode removeElements(ListNode head, int val) {
        if (head == null) return null;
        head.next = removeElements(head.next, val);
        return head.val == val ? head.next : head;
}

ODD EVEN 链表 - 单数index在前，双数在后

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* oddEvenList(ListNode* head) {
        
        if(!head)
            return NULL;
        if(!head->next || !head->next->next)
            return head;
        
        ListNode *cur = head;
        ListNode *eve = new ListNode();
        ListNode *eh = eve;
        ListNode *oh = cur;
  
        
        while(cur->next!=NULL)
        {

                
            ListNode *tmp = cur->next;
            
            if(cur->next->next)
                cur->next = cur->next->next;
            else
                cur->next = NULL;
            
            tmp->next = NULL;
            

            eve->next = tmp;
                
          
            
            eve = eve->next;    
            
            if(cur->next!=NULL)
                cur = cur->next;
            else
                break;

        }

        

        eh = eh->next; 
        cur->next = eh;
        
      
        return oh;
  
    }
};

ODD EVEN 链表 - 单数index在前，双数在后 简单写法

    ListNode* oddEvenList(ListNode* head) 
    {
        if(!head) return head;
        ListNode *odd=head, *evenhead=head->next, *even = evenhead;
        while(even && even->next)
        {
            odd->next = odd->next->next;
            even->next = even->next->next;
            odd = odd->next;
            even = even->next;
        }
        odd->next = evenhead;
        return head;

ODD EVEN 思想

odd 头就是head， even头是head->next, 目标是odd删除完后把even接在后面

当然，因为要遍历删除odd,肯定在最后一个节点停下，在这后面接上even的头就行了

回文递归版

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        ListNode *tail = head;
        return isPalindrome(tail,head);
    }
    
    
    bool isPalindrome(ListNode* tail,ListNode* &head){
        
        if(tail==NULL)
            return true;
        
        bool f = isPalindrome(tail->next,head);
        
        if(f==false)
            return false;
        
        if(tail->val != head->val)
            return false;
       
        
        head = head->next;
        
        return f;
    }
};

合并两个排过序的链表

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {

        
        ListNode *l3 = new ListNode(0);
       
        ListNode *head = l3;
        
        while(l1 && l2)
        {
            if(l1->val > l2->val)
            {
                l3->next = l2;
                l2 = l2->next;
                
            }
            else
            {
                l3->next = l1;
                l1 = l1->next;
                
            }
            
            l3 = l3->next;
        }

        l3->next = l1 ? l1 : l2;
        return head->next;
        
    }
};
