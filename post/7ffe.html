数组找index使左边所有元素之和=右边之和

class Solution {
public:
    int pivotIndex(vector<int>& nums) {
        
        if(nums.size()==0)
            return -1;
        
        if(nums.size()==1)
            return 0;
        
        int sum = 0;
        for(int i=0;i<nums.size();i++)
        {
            sum+=nums[i];
        }
        
        int csum=0;
        
        for(int i=0;i<nums.size();i++)
        {
            if(csum== sum-csum-nums[i])
                return i;
            
            csum+= nums[i];
        }
        
        return -1;
        
    }
};

找最大数的index并且这个数大于等于第二大的数的二倍

class Solution {
public:
    int dominantIndex(vector<int>& nums) {
        
        int max = 0;
        int sub_max = 0;
        
        int mi = 0;
        int ms =0;
        
        if(nums.size()==0){return -1;}
        if(nums.size()==1){return 0;}
            
        if(nums[0]>nums[1])
            {
                max = nums[0];
                mi = 0;
                sub_max=nums[1];
                ms = 1;
                
            }
        else
            {
                max = nums[1];
                mi = 1;
                sub_max=nums[0];
                ms = 0;
            }
            
        for(int i=2;i<nums.size();i++)
        {
            if(nums[i]<max && nums[i]>sub_max)
            {
                sub_max = nums[i];
                ms = i;
            }
            else if(nums[i]>max)
            {
                sub_max = max;
                ms = mi;
                
                max = nums[i];
                mi = i;
                
            
            }
        }
        
        if(max >= 2*sub_max)
            return mi;
        else
            return -1;
            
        
    }
};

Diagonal Traverse

class Solution {
public:
    vector<int> findDiagonalOrder(vector<vector<int>>& matrix) {
        
        
        int row = matrix.size();
        if(row==0)
            return vector<int>();
        int col = matrix[0].size();
        int total = row * col;
        

        
        vector<int> res(total);
        int i=0,j=0;
        
        int flag = 0;
        res[0] = matrix[0][0];
        
        for(int t=1;t<total;t++)
        {
            if(flag==0)
            {
                if(i-1>=0 && j+1<col)
                {   
                    i--;
                    j++;
                    res[t] = matrix[i][j];
                }
                else
                {
                    flag = 1;
                    
                    if(j+1<col)
                    {
                        j++;
                        res[t]=matrix[i][j];
                    }
                    else
                    {
                        i++;
                        res[t]=matrix[i][j];
                    }
                }
                
                
            }
            else
            {
                if(i+1<row && j-1>=0)
                {   
                    i++;
                    j--;
                    res[t] = matrix[i][j];
                }
                else
                {
                    flag = 0;
                    
                    if(i+1<row)
                    {
                        i++;
                        res[t]=matrix[i][j];
                    }
                    else
                    {
                        j++;
                        res[t]=matrix[i][j];
                    }
                }
            }
            
        
        }
        
        return res;
    }
};
