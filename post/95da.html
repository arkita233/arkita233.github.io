两两反转链表元素 - 递归

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        
        if(head==NULL){return NULL;}
        
        if(head->next == NULL){return head;}
        
        ListNode *tmp = head->next;
        head->next = swapPairs(tmp->next);
        tmp->next=head;
        
        return tmp;
        
    }
};

search data in binary tree - return node - recursion

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* searchBST(TreeNode* root, int val) {
        
        if(root){
        
        if(root->val == val)
            return root;
        
        if(val < root->val)
            return searchBST(root->left,val);
        if(val > root->val)
            return searchBST(root->right,val);
        }
        
        
        return NULL;
        
        
    }
};

递归打印杨辉三角的N行，N从0开始

终于自己设计了一波orz

class Solution {
public:
    vector<int> getRow(int rowIndex) {
        
        if(rowIndex == 0){
            vector<int> v(1);
            v[0]=1;
            return v;
        }


            
            
        vector<int> res = getRow(rowIndex-1);
        
       
        vector<int> v(rowIndex+1);
        
        for(int i=0;i<rowIndex+1;i++)
        {
            if(i==0)
                v[i]=1;
            else if(i==rowIndex)
                v[i]=1;
            else
            {
                v[i] = res[i-1] + res[i];
            }
        }
        
        return v;

        
    }
};

斐波那契 - 递归 - 重复计算费时间

class Solution {
public:
    int fib(int N) {
        if(N<2)
            return N;
        else
            return fib(N-2)+fib(N-1);
    }
};

斐波那契 - 保存前两个数版 - 省时

class Solution {
public:
    int fib(int N) {
        
        if(N<2)
            return N;
        
        int f = 0;
        int s = 1;
        int tmp = 0;
        
        for(int i=1;i<N;i++)
        {
            tmp = f + s;
            f = s;
            s = tmp;
        }
        
        return tmp;
    }
};

爬楼梯，每次能1阶或2阶，给层数求总路径数量 - 斐波那契

第三层开始，第n层分解为n-1层的路径+n-2层的路径

class Solution {
public:
    int climbStairs(int n) {
                
        if(n<=2)
            return n;
        
        int f = 1;
        int s = 2;
        int tmp = 0;
        
        for(int i=2;i<n;i++)
        {
            tmp = f + s;
            f = s;
            s = tmp;
        }
        
        return tmp;
    }
};

递归二叉树深度返回

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int maxDepth(TreeNode* root) {
        
        if(root==NULL)
            return NULL;
        
        return max(1+maxDepth(root->left),1+maxDepth(root->right));
        

        
    }
};

递归实现乘方

class Solution {
public:
    double myPow(double x, int n) {
        
        if(n==0)
            return 1;
        if(n==1)
            return x;
        if(n==-1)
            return 1/x;
        
        if(n%2 && n>0)
            return x*myPow(x*x,n/2);
        else if(n%2 && n<0)
             return 1/x*myPow(x*x,n/2);
        else if(n%2==0 && n<0)
            return myPow(x*x,n/2);
        else
            return myPow(x*x,n/2);
    }
};
