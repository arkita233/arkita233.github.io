1290 E - 0,1 链表转为十进制 例子 1-0-1 = 5

递归

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    void helper(ListNode* head, int& c, int& val)
    {
        if(!head){return;}
        
        helper(head->next,c,val);
        
        if(head->val==1)
            val += pow(2,c);
        
        c++;
        
    }
    
    int getDecimalValue(ListNode* head) {
        
        if(head==NULL){return 0;}
        int c = 0;
        int val=0;
        
        helper(head,c,val);
        
        return val;
        
    }
};

位运算 - 每次完成后整体左移=x2 0101 =5 1010=10，然后最后一位判断是否加1

class Solution {
public:
    int getDecimalValue(ListNode* head) {
        int ret = 0;
        while(head)
        {
            ret <<= 1;
            ret |= head->val;
            head = head->next;
        }
        return ret;
    }
};

876 E 链表找中点

双指针

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
    ListNode* middleNode(ListNode* head) {
        ListNode *slow = head, *fast = head;
        while (fast && fast->next)
            slow = slow->next, fast = fast->next->next;
        return slow;
    }

E 83 排过序的链表删除重复的节点

下一个重复就删，删到不重复后更新cur开始下一轮查重

class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        
        ListNode* cur = head;
        
        while(cur && cur->next)
        {
            
            while(cur->val==cur->next->val){
                if(cur->next->next)
                    cur->next = cur->next->next;
                else{
                    cur->next = NULL;
                    break;
                
                }
            }
            cur = cur->next;
        }
        
        return head;
    }
};

N 1019. Next Greater Node In Linked List 找这个节点之后比他大的一个节点，返回一个数组，找不到则是0

暴力解法

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        
        if(!head || !head->next){return vector<int>{0};}

            
        ListNode *cur = head;
        ListNode *net = cur;
        
        vector<int> res;
        
        while(cur)
        {
            
            net = cur->next;
            
            while(net && cur->val>=net->val)
            {
                net = net->next;
            }
            
            if(!net)
            {
                res.push_back(0);
            }
            else
            {
                res.push_back(net->val);
            }
            
            
            cur = cur->next;
        
                
        }
        
        return res;
    }
};

单调栈 - 查找前后第一个大于/小于的数

    vector<int> nextLargerNodes(ListNode* head) {
        
        vector<int> res;
        stack<int> s;
        
        for(auto h=head;h!=NULL;h=h->next)
            res.push_back(h->val);
        
        
        for(int i=res.size()-1;i>=0;i--)
        {
            
            int val = res[i];
            
            while(!s.empty() && s.top()<=res[i])
                s.pop();
            
            res[i]=s.empty()?0:s.top();
            
            s.push(val);
      
            
        }
            
        return res;
    }
