938 E Range Sum of BST是否存在一个路径，让节点之和=某一值

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool hasPathSum(TreeNode* root, int sum) {
        
        if(root==NULL){return false;}
        
        sum-=root->val;
        
        if(sum==0 && root->left==NULL && root->right==NULL)
            return true;

        
        return hasPathSum(root->left,sum) || hasPathSum(root->right,sum);
        

        
    }
};

617 E Merge Two Binary Trees 合并两棵树 - 值相加

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* mergeTrees(TreeNode* t1, TreeNode* t2) {
        
        if(t1 && t2){
            t1->val+=t2->val;
            t1->left = mergeTrees(t1->left,t2->left);
            t1->right = mergeTrees(t1->right,t2->right);
            return t1;
        }
        else
            return t1?t1:t2;
        
    }
};

589 E N-ary Tree Preorder Traversal N叉树前序遍历

递归

/*
// Definition for a Node.
class Node {
public:
    int val;
    vector<Node*> children;

    Node() {}

    Node(int _val) {
        val = _val;
    }

    Node(int _val, vector<Node*> _children) {
        val = _val;
        children = _children;
    }
};
*/

class Solution {
public:
    void helper(Node* root, vector<int>& v)
    {
        if(!root){return;}
        
        
        v.push_back(root->val);
        

        
        for(int i=0;i<root->children.size();i++)
        {
            helper(root->children[i],v);


        }
            
               
    }
        
    
    
    vector<int> preorder(Node* root) {
        
        vector<int> res;
        helper(root,res);
        
        return res;
        
    }
};

stack 版

    vector<int> preorder(Node* root) {
        
        vector<int> res;
        stack<Node*> s;
        
        if(!root){return res;}
        
        s.push(root);
        
        while(!s.empty())
        {
            Node* tmp = s.top();
            s.pop();
            res.push_back(tmp->val);
            
            for(int i=tmp->children.size()-1; i>=0;i--)
                s.push(tmp->children[i]);
                
        }
        
        return res;
        
    }

class Solution {
public:
    void helper(Node* root, vector<int>& v)
    {
        if(!root){return;}
        
        
        if(root->children.size()==0){ v.push_back(root->val);}
        
        else{
        
                for(int i=0;i<root->children.size();i++)
                {
                    helper(root->children[i],v);


                }
            
                v.push_back(root->val);
            }
        
    }
    
    
    vector<int> postorder(Node* root) {
        
        vector<int> res;
        helper(root,res);
        
        return res;
    }
};

stack 版

    vector<int> postorder(Node* root) {
        
        if(!root){return vector<int>();}
        
        stack<Node*> s;
        vector<int> res;
        
        s.push(root);
        
        while(!s.empty())
        {
            
            Node* tmp = s.top();
            
            res.push_back(tmp->val);
            
            s.pop();
            
            for(int i=-0;i<tmp->children.size();i++)
                s.push(tmp->children[i]);
            
            
                
        }
        
        reverse(res.begin(),res.end());
        
        return res;
        
    }

897. E Increasing Order Search Tree 中序遍历

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* res;
    TreeNode* cur;
    
    void helper(TreeNode* root)
    {
        
        if(!root){return;}
        
        
    
        
        helper(root->left);
        
        
        if(!res)
        {
            res = new TreeNode(root->val);
            cur = res;
        }
        else
        {
            cur->right = new TreeNode(root->val);
            cur = cur->right;
        }

        helper(root->right);
    }
    
    TreeNode* increasingBST(TreeNode* root) {
        
        if(!root){return root;}
        
        
        
        helper(root);
        
        return res;
        
    }
};

559. Maximum Depth of N-ary Tree

递归 - 返回上层+1，遍历同层child 取max深度 递-求孩子节点深度 归-+1并比较同层

    int maxDepth(Node* root) {
        
        if(!root){return 0;}
        int res = 0;
        
        for(int k=0;k<root->children.size();k++)
            res = max(res,maxDepth(root->children[k]));
        
        return res+1;
        
    }

遍历 - 用队列，每层进队，全pop出来一次深度++，每次循环pop一个时再把children入队开始下一层

queue<Node*> q;
int maxDepth(Node* root) {
    if(root == NULL) return 0; // 空节点筛除
    
    int res = 0;
    q.push(root);
    while(q.size()){ // 当该层还有节点
        for(int i = q.size(); i > 0; i--){ // 遍历该层所有节点
            Node* t = q.front(); // 获得队列头元素
            q.pop();             // 弹出队列头元素
            // 向队列尾插入队列头元素所有子节点
            for(int j = 0; j < t->children.size(); j++){
                q.push(t->children[j]);
            }
        }
        // 层数++
        res++;
    }
    
    return res;
}

    bool isUnivalTree(TreeNode* root) {

        int v = root->val;
        
        queue<TreeNode*> s;
        s.push(root);
        
        while(!s.empty())

左右翻转二叉树 E 226. Invert Binary Tree

    TreeNode* invertTree(TreeNode* root) {
        
        if(!root){return NULL;}
        
        TreeNode* tmp = root->left;
        
        root->left = root->right;
        root->right = tmp;
        
        invertTree(root->left);
        invertTree(root->right);
        
        
        return root;
        
        
    }

stack

TreeNode* invertTree(TreeNode* root) {
    std::stack<TreeNode*> stk;
    stk.push(root);
    
    while (!stk.empty()) {
        TreeNode* p = stk.top();
        stk.pop();
        if (p) {
            stk.push(p->left);
            stk.push(p->right);
            std::swap(p->left, p->right);
        }
    }
    return root;
}

637. E Average of Levels in Binary Tree 二叉树每层平均值

BFS - QUEUE

    vector<double> averageOfLevels(TreeNode* root) {
        
        if(!root){return vector<double>();}
        
        queue<TreeNode*> q; 
        vector<double> res;
        
        q.push(root);
  
        while(!q.empty())
        {
            double sum = 0;
            int s = q.size();
            
            for(int i=0;i<s;i++)
            {
                TreeNode* tmp = q.front();
                q.pop();
                
                if(tmp)
                    sum+=tmp->val;
                
                if(tmp->left)
                    q.push(tmp->left);
                if(tmp->right)
                    q.push(tmp->right);
            }
            
            
            
            res.push_back(sum/s);
        }
        
        return res;
    }

111. E Minimum Depth of Binary Tree

BFS - 每层+1，每层遍历到最后一个叶子的话直接return

    int minDepth(TreeNode* root) {
        
        if(!root){return 0;}
        
        queue<TreeNode*> q;
        int d = 1;
        
        q.push(root);
        
        while(!q.empty())
        {

            int s = q.size();
            
            for(int i=0;i<s;i++)
            {
                TreeNode* tmp = q.front();
                q.pop();
                
                if(tmp->left==NULL && tmp->right==NULL)
                    return d;
                
                if(tmp->left)
                    q.push(tmp->left);
                if(tmp->right)
                    q.push(tmp->right);
            }
            
            d++;
        }
        
        return d;
    }

递归

int minDepth(TreeNode* root) {
    if (!root) return 0;
    int L = minDepth(root->left), R = minDepth(root->right);
    return 1 + (min(L, R) ? min(L, R) : max(L, R));
}
